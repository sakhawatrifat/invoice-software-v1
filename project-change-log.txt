
INSERT INTO users (
    user_type,
    is_automation_chatbot,
    uid,
    image,
    name,
    email,
    password,
    email_verified_at,
    device_token,
    ip_address,
    status,
    created_at,
    updated_at
) 
SELECT 
    'admin',
    1,
    CONCAT('chatbot-', REPLACE(UUID(), '-', '')),
    NULL,
    'Automation Chatbot',
    'chatbot@system',
    -- use a bcrypt hash of a random string, e.g. from Laravel: password_hash('random32charstringhere', PASSWORD_BCRYPT)
    '$2y$12$' || SUBSTRING(MD5(RAND()) FROM 1 FOR 53),
    NOW(),
    NULL,
    NULL,
    'Active',
    NOW(),
    NOW()
FROM DUAL
WHERE NOT EXISTS (
    SELECT 1 FROM users WHERE email = 'chatbot@system.local' LIMIT 1
);
